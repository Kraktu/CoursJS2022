Chapitre 3 - Exercice 10 : Chante Sloubi

Factorisez le code suivant :


import java.util.Scanner;
public class exoFacto {
public static Scanner input = new Scanner(System.in);

public static void main(String[] args) {

System.out.println("Une partie de Sloubi");
for (int i = 0; i < 20; i++) {
System.out.println("Sloubi "+ i);
}

System.out.println("let's go");
int lim = (int)Math.floor(Math.random()*(50-15+1)+15);
for (int i = 20; i < 20+lim; i++) {
System.out.println("Sloubi "+ i);
}

System.out.println("Encore!");
for (int i = 20+lim; i < 20+lim+24; i++) {
System.out.println("Sloubi "+ i);
}

System.out.println("C'est reparti");
System.out.println("Jusque combien vous voulez aller cette fois-ci ?");
int lim2 = input.nextInt();

for (int i = 20+24+lim; i < 20+lim+24+lim2; i++) {
System.out.println("Sloubi "+ i);
}

System.out.println("On continue !");
for (int i = 20+lim+24+lim2; i < 20+lim+24+lim2+73; i++) {
System.out.println("Sloubi "+ i);
}
}
}
